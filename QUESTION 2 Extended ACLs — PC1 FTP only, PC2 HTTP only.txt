■ STEPS

Step 1 — Configure IPs on Router
--R1(config)#
interface gig0/0
ip address 172.22.34.65 255.255.255.224
no shutdown
exit
interface gig0/1
ip address 172.22.34.97 255.255.255.240
no shutdown
exit
interface gig0/2
ip address 172.22.34.1 255.255.255.192
no shutdown
exit

Step 2 — Configure ACL for PC1 (FTP + ICMP only)
--R1(config)#
access-list 100 permit tcp 172.22.34.64 0.0.0.31 host 172.22.34.62 eq ftp
access-list 100 permit icmp 172.22.34.64 0.0.0.31 host 172.22.34.62
interface gig0/0
ip access-group 100 in
exit

Step 3 — Configure Named ACL for PC2 (HTTP + ICMP only)
--R1(config)#
ip access-list extended HTTP_ONLY
permit tcp 172.22.34.96 0.0.0.15 host 172.22.34.62 eq www
permit icmp 172.22.34.96 0.0.0.15 host 172.22.34.62
exit
interface gig0/1
ip access-group HTTP_ONLY in
exit


Step 4 — VERIFY
---PC1 command prompt

http://172.22.34.62   //it will be request timeout
---PC2 command prompt
